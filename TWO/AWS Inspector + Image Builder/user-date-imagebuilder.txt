#!/bin/bash

# This command help us to enable repositorys for yum install packages
# amazon-linux-extras
# amazon-linux-extras enable package
# EX: amazon-linux-extras enable redis6 && yum install -y redis


# Update specific PACKAGE with yum:
# yum update -y packagename


# Update the SYSTEM PACKAGES
sudo yum update -y

# Installing SSM Agent on Instance:
sudo yum install -y https://s3.amazonaws.com/ec2-downloads-windows/SSMAgent/latest/linux_amd64/amazon-ssm-agent.rpm
sudo systemctl restart amazon-ssm-agent

# Installing Inspector Classic Agent on Instance:
cd /home/ec2-user/
sudo wget https://inspector-agent.amazonaws.com/linux/latest/install
sudo bash install
